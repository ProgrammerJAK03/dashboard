Common Reasons Why Data Isn't Saving to Google Sheets

  1. Incorrect Script URL - Make sure you've deployed the Google Apps Script as a web app and copied the correct URL
  2. Authorization Required - You may need to authorize the script to run, even if deployed to run as "Anyone"
  3. Google Sheets Permissions - Ensure the Google Sheet has edit permissions for anyone with the link
  4. CORS Issues - Google Apps Script web apps sometimes have Cross-Origin Resource Sharing limitations
  5. Script Errors - There might be an error in your Google Apps Script code

Updated Google Apps Script Code

Replace your Google Apps Script code with this updated version:

// Replace with your Sheet ID
var SHEET_ID = "your_sheet_id_here";

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    
    // Check if sheet is empty and add headers if needed
    if (sheet.getLastRow() === 0) {
      var headers = ["Date", "Registration", "ChasisNumber", "CustomerName", "ShowroomName", "ReturnFile", "NP"];
      sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    }
    
    // Add new row with data
    var newRow = [
      data.date,
      data.registration,
      data.chasis,
      data.customer,
      data.showroom,
      data.returnFile,
      data.np
    ];
    
    sheet.appendRow(newRow);
    
    return ContentService
      .createTextOutput(JSON.stringify({result: "success", row: newRow}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({result: "error", error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  try {
    var sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
    
    // Check if sheet is empty
    if (sheet.getLastRow() === 0) {
      return ContentService
        .createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    var data = sheet.getDataRange().getValues();
    
    // If there's no data except headers
    if (data.length <= 1) {
      return ContentService
        .createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    // Convert to array of objects
    var headers = data[0];
    var result = [];
    
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var obj = {};
      
      for (var j = 0; j < headers.length; j++) {
        obj[headers[j].toLowerCase()] = row[j];
      }
      
      result.push(obj);
    }
    
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({result: "error", error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Deployment Instructions

  1. Open your Google Apps Script project
  2. Click "Deploy" > "New deployment"
  3. Select type "Web app"
  4. Set "Execute as" to "Me"
  5. Set "Who has access" to "Anyone"
  6. Click "Deploy"
  7. Copy the web app URL and paste it in the Debug section above
  8. Click "Save URL"
  9. Authorize the script when prompted
